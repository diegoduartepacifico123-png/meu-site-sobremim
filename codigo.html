JS Básico (Criação da Página) 
// ------------------------
// codigo.js - Interatividade da Página
// ------------------------

document.addEventListener('DOMContentLoaded', function () {

  // 1) Botão "mágico" para mostrar/esconder código
  const btnMostrarCodigo = document.querySelector('.btn-mostrar-codigo');
  const codigoEscondido = document.querySelector('.codigo-escondido');

  btnMostrarCodigo.addEventListener('click', function () {
    if (codigoEscondido.classList.contains('show')) {
      codigoEscondido.classList.remove('show');
      btnMostrarCodigo.textContent = 'Mostrar Código';
    } else {
      codigoEscondido.classList.add('show');
      btnMostrarCodigo.textContent = 'Esconder Código';
    }
  });

  // 2) Efeito neon nas linhas de código ao passar o mouse
  const linhasCodigo = codigoEscondido.querySelectorAll('pre, code');

  linhasCodigo.forEach(function (linha) {
    linha.addEventListener('mouseenter', function () {
      linha.style.textShadow = '0 0 8px #00ff88, 0 0 12px #00b8ff';
      linha.style.color = '#00ff88';
    });
    linha.addEventListener('mouseleave', function () {
      linha.style.textShadow = 'none';
      linha.style.color = '#00ff88';
    });
  });

  // 3) Aviso divertido para a seção de novidades
  const novidades = document.querySelector('.novidades');
  if (novidades) {
    novidades.addEventListener('click', function () {
      alert('🚀 Novos ensinamentos serão adicionados em breve! Fique atento!');
    });
  }
});
